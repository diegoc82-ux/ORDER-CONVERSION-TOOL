// ── Parsing ───────────────────────────────────────────────────────────────
async function parseOrder() {
  hideMessages();
  document.getElementById('results').style.display = 'none';

  const btn = document.getElementById('parse-btn');
  btn.disabled = true;
  document.getElementById('spinner').style.display = 'block';

  try {
    let text = '';

    if (activeTab === 'file') {
      if (!selectedFile) { showError('Please select a PDF file first.'); return; }
      const form = new FormData();
      form.append('file', selectedFile);
      const res = await fetch('/api/extract-pdf', { method:'POST', body: form });
      const data = await res.json();
      if (!res.ok) { showError(data.error || 'PDF extraction failed.'); return; }
      text = data.text;
    } else {
      text = document.getElementById('text-input').value.trim();
      if (!text) { showError('Please paste the purchase order text.'); return; }
    }

    const parsed = parsePOText(text);
    renderForm(parsed);
  } finally {
    document.getElementById('spinner').style.display = 'none';
    btn.disabled = false;
  }
}

// ── PO Text Parser ────────────────────────────────────────────────────────
function parsePOText(text) {
  const lines = text.split('\n').map(l => l.trim());

  // ── PO Number ────────────────────────────────────────────────────────────
  let poNumber = '';
  const poM =
    text.match(/P\.?O\.?\s*NO\.?\s+(\w+)/i)   ||
    text.match(/P\.?O\.?\s*#\s*(\w+)/i)        ||
    text.match(/PO[_\-\s]?(\d{4,6})/i);
  if (poM) {
    const raw = poM[1].replace(/^PO_?/i, '');
    poNumber = 'PO_' + raw.padStart(5, '0');
  }

  // ── Order date ───────────────────────────────────────────────────────────
  let orderDate = '';
  const dateM =
    text.match(/DATE\s+(\d{1,2}[\/\-]\d{1,2}[\/\-]\d{4})/i) ||
    text.match(/Date[:\s]+(\d{1,2}[\/\-]\d{1,2}[\/\-]\d{4})/i) ||
    text.match(/(\d{1,2}[\/\-]\d{1,2}[\/\-]\d{4})/);
  if (dateM) orderDate = dateM[1].replace(/-/g, '/');

  // ── Delivery date ───────────────────────────────────────────────────────
  let deliveryDate = '';
  const delivM = text.match(/(?:ship(?:ping)?\s*date|deliver(?:y)?\s*(?:date|by)|required\s*(?:by|date))[:\s]+(\d{1,2}[\/\-]\d{1,2}[\/\-]\d{4})/i);
  if (delivM) deliveryDate = delivM[1].replace(/-/g, '/');

  // ── Ship To / client info ─────────────────────────────────────────────────
  let clientName = '', address1 = '', address2 = '';

  // Method A: "VENDOR ... SHIP TO" combined header (two-column layout, e.g. QuickBooks POs)
  // Format: next line has "Our Vendor Name  Client Name  DATE XX/XX/XXXX"
  const vendorShipIdx = lines.findIndex(l => /VENDOR.+SHIP\s*TO/i.test(l));
  if (vendorShipIdx >= 0) {
    const infoLine = lines[vendorShipIdx + 1] || '';
    // Strip our company name and the DATE portion to isolate client name
    let clientPart = infoLine
      .replace(/Ultra\s*1?\s*Plus\b/gi, '')
      .replace(/DATE\s+[\d\/\-]+/i, '')
      .trim();
    if (clientPart) clientName = clientPart;

    // Subsequent lines: address continuation or more company name
    for (let k = vendorShipIdx + 2; k <= Math.min(vendorShipIdx + 8, lines.length - 1); k++) {
      const l = lines[k];
      if (!l || /^(QTY|CE\s|CE\s+WAREHOUSE|CE\s+DFW|SHIP|AMOUNT)/i.test(l)) break;
      if (/^\d+\s+[A-Z]/i.test(l) && !address1) {
        address1 = l;          // line starting with number = street
      } else if (/[A-Z]{2}\s+\d{5}/.test(l) && !address2) {
        address2 = l;          // line with ST 00000 = city/state/zip
      } else if (!address1) {
        // No address found yet → this is continuation of company name
        clientName = (clientName + ' ' + l).trim();
      }
    }

  } else {
    // Method B: standalone "SHIP TO" line
    const stIdx = lines.findIndex(l => /^SHIP\s*TO\s*:?$/i.test(l));
    if (stIdx >= 0) {
      const block = [];
      for (let i = stIdx + 1; i < Math.min(stIdx + 8, lines.length); i++) {
        const l = lines[i];
        if (!l || /^(VENDOR|P\.O\.|DATE|QTY|RATE|AMOUNT|CE\s|SHIP)/i.test(l)) break;
        block.push(l);
      }
      if (block[0]) clientName = block[0];
      const streetLine = block.slice(1).find(l => /\d/.test(l));
      if (streetLine) address1 = streetLine;
      const cityLine = block.find(l => /[A-Z]{2}\s+\d{5}/.test(l));
      if (cityLine && cityLine !== address1) address2 = cityLine;
      else if (block[2]) address2 = block[2];
    }
  }

  // ── Line items ────────────────────────────────────────────────────────────
  // In many PO formats (QuickBooks) the qty appears on the FIRST line of the
  // item description — BEFORE the UL code — so we search BACKWARDS first.
  // Pattern: "Some description text  QTY  RATE  AMOUNT"
  // Note: QTY can include a thousands comma  e.g. "1,008 3.06 3,084.48"
  const QTY_LINE = /([\d,]+)\s+([\d,]+\.\d{2})\s+([\d,]+\.\d{2})\s*$/;

  const items = [];
  const seen  = new Set();

  for (let i = 0; i < lines.length; i++) {
    // Match UL codes — no $ so trailing text like "Tote" is allowed
    // e.g. UL802-265-Gallon Tote, UL206-55GAL, UL501-5GAl
    const skuM = lines[i].match(/^(UL\d{3,4})[\s\-]*([\w\-\.]+)/i);
    if (!skuM) continue;

    const code   = skuM[1].toUpperCase();
    const suffix = skuM[2];
    const mapped = mapPresentation(suffix, code);
    if (!mapped) continue;

    const key = code + '|' + mapped.pres;
    if (seen.has(key)) continue;
    seen.add(key);

    let rawQty = 0, unitPrice = 0, amount = 0;

    // ── 1. Look BACKWARDS (most common: qty is on the first description line) ──
    for (let j = i - 1; j >= Math.max(0, i - 8); j--) {
      const m = lines[j].match(QTY_LINE);
      if (m) {
        const q = parseInt(m[1].replace(/,/g, ''));   // strip thousands comma e.g. 1,008 → 1008
        const r = parseFloat(m[2].replace(/,/g, ''));
        // Sanity: qty ≥ 1, rate > 0 (avoids matching zip codes etc.)
        if (q > 0 && r > 0) {
          rawQty    = q;
          unitPrice = r;
          amount    = parseFloat(m[3].replace(/,/g, ''));
          break;
        }
      }
    }

    // ── 2. Look FORWARDS (fallback for formats that put qty after the code) ──
    if (!rawQty) {
      for (let j = i + 1; j <= Math.min(i + 3, lines.length - 1); j++) {
        const m = lines[j].match(/^(\d+)\s+([\d,]+\.?\d*)\s+([\d,]+\.?\d*)/);
        if (m) {
          rawQty    = parseInt(m[1]);
          unitPrice = parseFloat(m[2].replace(/,/g, ''));
          amount    = parseFloat(m[3].replace(/,/g, ''));
          break;
        }
        const qOnly = lines[j].match(/^(\d+)$/);
        if (qOnly) { rawQty = parseInt(qOnly[1]); break; }
      }
    }

    if (!rawQty) rawQty = 1;

    // ── Unit conversion ───────────────────────────────────────────────────
    let caseQty = rawQty;
    let poQtyLabel = '';
    if (mapped.unitsPerCase && mapped.unitsPerCase > 1) {
      caseQty    = Math.ceil(rawQty / mapped.unitsPerCase);
      poQtyLabel = rawQty + ' ' + mapped.poUnit;
    } else if (mapped.unitsPerCase === 1 && mapped.poUnit) {
      poQtyLabel = rawQty + ' ' + mapped.poUnit;
    }

    items.push({
      ul_code:        code,
      presentation:   mapped.pres,
      qty:            caseQty,
      po_qty_label:   poQtyLabel,
      unit_price:     unitPrice,
      amount,
      units_per_case: mapped.unitsPerCase ?? null,
    });
  }

  return { poNumber, orderDate, deliveryDate, clientName, address1, address2, items };
}

// mapPresentation returns { pres, poUnit, unitsPerCase }
// poUnit/unitsPerCase are set when the customer ordered in individual units
// (e.g. QT, GAL) so we can show the conversion and store original PO qty.
// ulCode is used to distinguish coolant/WW (UL9xx) from oil (all others)
// for the ambiguous "GAL" suffix → BOX (6G) vs BOX (4G).
function mapPresentation(suffix, ulCode = '') {
  const s = suffix.toUpperCase().replace(/\s+/g, '');
  // UL900–UL999 = coolants & windshield washer fluid → BOX (6G) / 6 per case
  const isCoolant = /^UL9\d{2}$/i.test(ulCode.trim());

  // ── Bulk / drums / pails ─────────────────────────────────────────────────
  if (/55G(AL)?$/.test(s))            return { pres: 'DRUM (55G)',      poUnit: null, unitsPerCase: null };
  if (/^5G(AL|AI)?$/.test(s))         return { pres: 'PAIL (5G)',       poUnit: null, unitsPerCase: null };
  if (/265/.test(s))                  return { pres: 'TOTE (265G)',     poUnit: null, unitsPerCase: null };
  if (/250/.test(s))                  return { pres: 'TOTE (250G)',     poUnit: null, unitsPerCase: null };
  if (/330/.test(s))                  return { pres: 'TOTE (330G)',     poUnit: null, unitsPerCase: null };

  // ── Box / case formats — customer ordered by full CASE already ───────────
  if (/^12Q/.test(s))                            return { pres: 'BOX (12Q)',      poUnit: null, unitsPerCase: null };
  if (/^2X2\.5G?|^2-2\.5G?/.test(s))            return { pres: 'BOX (2 X 2.5G)', poUnit: '2.5G', unitsPerCase: 2 };
  if (/^1X2\.5G?|^1-2\.5G?/.test(s))            return { pres: 'BOX (1 X 2.5G)', poUnit: null, unitsPerCase: null };
  if (/^4G(AL)?$/.test(s))                       return { pres: 'BOX (4G)',       poUnit: null, unitsPerCase: null };
  if (/^6G(AL)?$/.test(s))                       return { pres: 'BOX (6G)',       poUnit: null, unitsPerCase: null };
  if (/^3[\/\-]5QT/.test(s))                     return { pres: 'BOX (3/5QTS)',   poUnit: null, unitsPerCase: null };

  // ── Unit-based ordering — customer ordered in individual units ────────────
  // QT / 1QT  →  BOX (12Q),  divide by 12
  if (/^1?QT$/.test(s))                          return { pres: 'BOX (12Q)',      poUnit: 'QT',  unitsPerCase: 12 };
  // 5QT  →  BOX (3/5QTS),  divide by 3
  if (/^5QT$/.test(s))                           return { pres: 'BOX (3/5QTS)',   poUnit: '5QT', unitsPerCase:  3 };
  // 1GAL / GAL  →  BOX (6G) for coolants & WW (UL9xx), BOX (4G) for oils
  if (/^1?GAL$/.test(s))  return isCoolant
    ? { pres: 'BOX (6G)', poUnit: 'GAL', unitsPerCase: 6 }
    : { pres: 'BOX (4G)', poUnit: 'GAL', unitsPerCase: 4 };
  // 2.5G / 2.5GAL  →  BOX (1 x 2.5G),  divide by 1
  if (/^2\.5G(AL)?$/.test(s))                    return { pres: 'BOX (1 X 2.5G)', poUnit: '2.5G', unitsPerCase: 1 };

  // ── Other ────────────────────────────────────────────────────────────────
  if (/JC$|JERRCAN|JERRYCAN/.test(s))            return { pres: 'JERRYCAN (20L)', poUnit: null, unitsPerCase: null };
  if (/CASE/.test(s))                            return { pres: 'CASE 10/1',      poUnit: null, unitsPerCase: null };
  if (/KEG/.test(s))                             return { pres: 'KEG (120LBS)',   poUnit: null, unitsPerCase: null };
  return null;
}

// ── Calculations ──────────────────────────────────────────────────────────
function calcItem(code, presentation, qty) {
  const calc = PRES_CALC[presentation] || {};
  const dispatchSpec = calc.dispatch_spec || presentation;
  const unitQty   = qty * (calc.unit_mult || 1);
  const palletDiv = calc.pallet_div;
  const palletQty = palletDiv ? +(unitQty / palletDiv).toFixed(2) : null;
  const estWeight = palletQty !== null ? +(palletQty * (calc.weight_mult || 0)).toFixed(2) : null;

  const product = catalog[code] || {};
  const sku     = (product.presentations || {})[dispatchSpec] || '';
  const desc    = product.name || '';

  return { dispatchSpec, unitQty, palletQty, estWeight, sku, desc };
}

// ── Form rendering ────────────────────────────────────────────────────────
function renderForm(parsed) {
  document.getElementById('f-client').value   = parsed.clientName  || '';
  document.getElementById('f-po').value       = parsed.poNumber    || '';
  document.getElementById('f-addr1').value    = parsed.address1    || '';
  document.getElementById('f-addr2').value    = parsed.address2    || '';
  document.getElementById('f-date').value     = parsed.orderDate   || '';
  document.getElementById('f-delivery').value = parsed.deliveryDate|| '';

  hasTotes = parsed.items.some(it => (PRES_CALC[it.presentation]||{}).tote);
  toteReturn = false;

  const prompt = document.getElementById('tote-prompt');
  if (hasTotes) {
    document.getElementById('tote-client').textContent = parsed.clientName || 'this client';
    prompt.style.display = 'block';
    document.getElementById('generate-btn').style.display = 'none';
  } else {
    prompt.style.display = 'none';
    document.getElementById('generate-btn').style.display = 'inline-block';
  }

  const tbody = document.getElementById('items-body');
  tbody.innerHTML = '';
  parsed.items.forEach(it => appendRow(it.ul_code, it.presentation, it.qty, it.po_qty_label || '', it.unit_price || 0, it.units_per_case ?? null));
  refreshTotals();

  document.getElementById('results').style.display = 'block';
  document.getElementById('results').scrollIntoView({ behavior: 'smooth' });
}

function appendRow(code, presentation, qty, poQtyLabel = '', unitPrice = 0, unitsPerCase = null) {
  const c = calcItem(code, presentation, qty);
  const tbody = document.getElementById('items-body');
  const tr = document.createElement('tr');
  tr.dataset.code = code;

  const presOptions = PRESENTATIONS.map(p =>
    `<option value="${p}" ${p===presentation?'selected':''}>${p}</option>`
  ).join('');

  // PO QTY cell: shows original ordered units if a conversion happened
  const poQtyHtml = poQtyLabel
    ? `<span style="color:#1a5c9a;font-weight:bold;font-size:.78rem;">${esc(poQtyLabel)}</span>`
    : `<span style="color:#aaa;font-size:.73rem;">—</span>`;

  // If customer ordered in units (e.g. QT, GAL), unitPrice is per-unit → convert to per-case
  const casePrice = unitPrice > 0 && unitsPerCase && unitsPerCase > 1
    ? +(unitPrice * unitsPerCase).toFixed(2)
    : unitPrice;
  const total = qty * casePrice;

  tr.innerHTML = `
    <td><input class="tbl-input" style="width:64px;text-transform:uppercase" value="${esc(code)}" onchange="onCodeChange(this)" oninput="this.value=this.value.toUpperCase()"></td>
    <td><select class="tbl-select" onchange="onPresChange(this)">${presOptions}</select></td>
    <td class="num"><input class="tbl-input cases-input" style="width:54px;text-align:right" type="number" min="1" value="${qty}" onchange="onRowChange(this)"></td>
    <td class="derived poqty-cell num">${poQtyHtml}</td>
    <td class="derived sku-cell" style="font-size:.75rem;color:#555;">${esc(c.sku)||'<span class="warn-txt">?</span>'}</td>
    <td class="derived desc-cell" style="font-size:.75rem;">${esc(c.desc)}</td>
    <td class="derived spec-cell" style="font-size:.75rem;"><code>${esc(c.dispatchSpec)}</code></td>
    <td class="derived uqty-cell num">${c.unitQty}</td>
    <td class="derived pqty-cell num">${c.palletQty !== null ? c.palletQty.toFixed(2) : '<span class="warn-txt">?</span>'}</td>
    <td class="derived ewt-cell num">${c.estWeight !== null ? c.estWeight.toLocaleString() : '<span class="warn-txt">?</span>'}</td>
    <td class="num"><input class="tbl-input price-input" style="width:68px;text-align:right" type="number" min="0" step="0.01" value="${casePrice > 0 ? casePrice.toFixed(2) : ''}" placeholder="0.00" onchange="onRowChange(this)"></td>
    <td class="derived total-cell num" data-raw="${total.toFixed(2)}">${total > 0 ? '$' + fmtMoney(total) : '<span style="color:#aaa">—</span>'}</td>
    <td><button class="btn-del" onclick="deleteRow(this)" title="Remove">&#x2715;</button></td>
  `;
  tbody.appendChild(tr);
}

function onCodeChange(input) {
  input.value = input.value.toUpperCase().trim();
  onRowChange(input);
}

// When presentation changes, recalculate PO QTY label from stored raw qty if possible
function onPresChange(select) {
  const tr = select.closest('tr');
  const pres = select.value;
  const calc = PRES_CALC[pres] || {};
  // If there's already a PO qty shown (blue text), recalculate cases from it
  const poCell = tr.querySelector('.poqty-cell');
  const existingLabel = poCell?.querySelector('span[style*="1a5c9a"]')?.textContent?.trim();
  if (existingLabel && calc.units_per_case) {
    const rawMatch = existingLabel.match(/^(\d+)/);
    if (rawMatch) {
      const rawQty = parseInt(rawMatch[1]);
      const newCaseQty = Math.ceil(rawQty / calc.units_per_case);
      tr.querySelector('input[type=number]').value = newCaseQty;
    }
  }
  onRowChange(select);
}

function onRowChange(el) {
  const tr = el.closest('tr');
  const code  = tr.querySelector('input:not([type=number])').value.trim().toUpperCase();
  const pres  = tr.querySelector('select').value;
  const qty   = parseInt(tr.querySelector('.cases-input').value) || 1;
  const price = parseFloat(tr.querySelector('.price-input').value) || 0;
  const total = qty * price;
  const c = calcItem(code, pres, qty);

  tr.querySelector('.sku-cell').innerHTML    = c.sku  ? esc(c.sku)  : '<span class="warn-txt">Not found</span>';
  tr.querySelector('.desc-cell').innerHTML   = esc(c.desc);
  tr.querySelector('.spec-cell').innerHTML   = `<code>${esc(c.dispatchSpec)}</code>`;
  tr.querySelector('.uqty-cell').textContent = c.unitQty;
  tr.querySelector('.pqty-cell').innerHTML   = c.palletQty !== null ? c.palletQty.toFixed(2) : '<span class="warn-txt">?</span>';
  tr.querySelector('.ewt-cell').innerHTML    = c.estWeight !== null ? c.estWeight.toLocaleString() : '<span class="warn-txt">?</span>';
  const totalCell = tr.querySelector('.total-cell');
  totalCell.dataset.raw = total.toFixed(2);
  totalCell.innerHTML   = total > 0 ? '$' + fmtMoney(total) : '<span style="color:#aaa">—</span>';

  refreshTotals();
  refreshTotePrompt();
}

function deleteRow(btn) {
  btn.closest('tr').remove();
  refreshTotals();
  refreshTotePrompt();
}

function addRow() {
  appendRow('', PRESENTATIONS[0], 1, '', 0, null);
  refreshTotals();
}

function recalcAll() {
  document.querySelectorAll('#items-body tr').forEach(tr => {
    const sel = tr.querySelector('select');
    if (sel) onRowChange(sel);
  });
  showOk('✅ All rows recalculated.');
}

function refreshTotals() {
  let totalP = 0, totalW = 0, totalAmt = 0;
  document.querySelectorAll('#items-body tr').forEach(tr => {
    const pqty = parseFloat(tr.querySelector('.pqty-cell').textContent) || 0;
    const ewt  = parseFloat(tr.querySelector('.ewt-cell').textContent.replace(/,/g,'')) || 0;
    const amt  = parseFloat(tr.querySelector('.total-cell')?.dataset.raw || 0);
    totalP += pqty; totalW += ewt; totalAmt += amt;
  });
  document.getElementById('total-pallets').textContent = totalP.toFixed(2);
  document.getElementById('total-weight').textContent  = totalW.toLocaleString();
  document.getElementById('total-amount').innerHTML    = totalAmt > 0 ? '<strong>$' + fmtMoney(totalAmt) + '</strong>' : '—';
}

function refreshTotePrompt() {
  hasTotes = false;
  document.querySelectorAll('#items-body tr').forEach(tr => {
    const pres = tr.querySelector('select')?.value;
    if (pres && (PRES_CALC[pres]||{}).tote) hasTotes = true;
  });
  if (!hasTotes) {
    document.getElementById('tote-prompt').style.display = 'none';
    document.getElementById('generate-btn').style.display = 'inline-block';
  }
}

}

// ── CSV Generation ────────────────────────────────────────────────────────
async function generateCSV() {
  const client = document.getElementById('f-client').value.trim();
  const po     = document.getElementById('f-po').value.trim();
  const addr1  = document.getElementById('f-addr1').value.trim();
  const addr2  = document.getElementById('f-addr2').value.trim();
  const date   = document.getElementById('f-date').value.trim();
  const delivery = document.getElementById('f-delivery').value.trim();

  if (!client || !po) { showError('Client name and PO Number are required.'); return; }

  // Fetch current counter
  let lastDispatch = 3086;
  try {
    const r = await fetch('/api/counter');
    const d = await r.json();
    lastDispatch = d.last;
  } catch(e) {}
  const dispatchNum = lastDispatch + 1;

  // Build rows
  const rows = [];
  rows.push(['Dispatch Note', '#' + dispatchNum, '', '', '', '', '', '']);
  rows.push(['Ultra1Plus',    '',                 '', '', '', '', '', '']);
  rows.push(['',              '',                 '', '', '', '', '', '']);
  rows.push(['SHIP TO:', '', '', '', 'Customer PO Number:', po, '', '']);
  rows.push([client,    '', '', '', 'Date of order:',       date, '', '']);
  rows.push([addr1,     '', '', '', 'Shipping date:',       delivery, '', '']);
  rows.push([addr2,     '', '', '', 'Shipping time:',       '', '', '']);
  rows.push(['',        '', '', '', '', '', '', '']);
  rows.push(['CODE', 'CASES (DISPATCH)', 'PO QTY (ORDERED)', 'SKU', 'DESCRIPTION', 'SPEC', 'UNIT QTY', 'PALLET QTY', 'EST WEIGHT', 'PRICE/CASE', 'TOTAL', 'PO']);

  let totalPallets = 0, totalWeight = 0;
  document.querySelectorAll('#items-body tr').forEach(tr => {
    const code  = tr.querySelector('input:not([type=number])').value.trim().toUpperCase();
    if (!code) return;
    const cases    = tr.querySelector('.cases-input').value.trim();
    const poQtyRaw = tr.querySelector('.poqty-cell span[style*="1a5c9a"]')?.textContent?.trim() || '';
    const sku      = tr.querySelector('.sku-cell').textContent.trim();
    const desc     = tr.querySelector('.desc-cell').textContent.trim();
    const spec     = tr.querySelector('.spec-cell').textContent.trim();
    const uqty     = tr.querySelector('.uqty-cell').textContent.trim();
    const pqty     = parseFloat(tr.querySelector('.pqty-cell').textContent) || 0;
    const ewt      = parseFloat(tr.querySelector('.ewt-cell').textContent.replace(/,/g,'')) || 0;
    const priceVal = parseFloat(tr.querySelector('.price-input').value) || 0;
    const totalVal = parseFloat(tr.querySelector('.total-cell').dataset.raw || 0);
    rows.push([code, cases, poQtyRaw, sku, desc, spec, uqty, pqty.toFixed(2), ewt.toLocaleString(), priceVal > 0 ? priceVal.toFixed(2) : '', totalVal > 0 ? totalVal.toFixed(2) : '', po]);
    totalPallets += pqty;
    totalWeight  += ewt;
  });

  let grandTotal = 0;
  document.querySelectorAll('#items-body tr').forEach(tr => {
    grandTotal += parseFloat(tr.querySelector('.total-cell')?.dataset.raw || 0);
  });
  rows.push(['', '', '', '', '', '', '', totalPallets.toFixed(2), totalWeight.toLocaleString(), '', grandTotal > 0 ? grandTotal.toFixed(2) : '', '']);
  rows.push(['', '', '', '', '', '', '', '', '', '', '', '']);

  const notes = toteReturn ? 'Tote Return Program applied — containers to be returned by customer' : '';
  rows.push(['NOTES:', notes, '', '', '', '', '', '', '', '', '', '']);

  // Save counter
  try { await fetch('/api/counter', { method:'POST', headers:{'Content-Type':'application/json'}, body: JSON.stringify({last: dispatchNum}) }); } catch(e){}
  updateDispatchBadge(dispatchNum + 1);

  // Build and download CSV
  const csvContent = '\uFEFF' + rows.map(r => r.map(csvCell).join(',')).join('\r\n');
  const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
  const url  = URL.createObjectURL(blob);
  const a    = document.createElement('a');
  a.href     = url;
  a.download = `Dispatch_${dispatchNum}_${po.replace(/[^\w]/g,'_')}.csv`;
  a.click();
  URL.revokeObjectURL(url);

  showOk('✅ Dispatch #' + dispatchNum + ' downloaded successfully!');
}

function csvCell(val) {
  const s = String(val ?? '');
  return (s.includes(',') || s.includes('"') || s.includes('\n')) ? '"' + s.replace(/"/g, '""') + '"' : s;
}
